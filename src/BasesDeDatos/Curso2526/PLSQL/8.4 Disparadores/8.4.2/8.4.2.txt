8. Actualizar precios.

SET SERVEROUTPUT ON;
CREATE OR REPLACE TRIGGER tr_8
BEFORE INSERT OR UPDATE OF precio ON libros
FOR EACH ROW
BEGIN
    IF :NEW.precio > 50 THEN
        :NEW.precio := ROUND(:NEW.precio);
    END IF;
END;
/

9. Actualizar precios mensajes.

--Apartado 1

SET SERVEROUTPUT ON;
CREATE OR REPLACE TRIGGER tr_9_apartado1
BEFORE INSERT OR UPDATE OF precio ON libros
FOR EACH ROW
BEGIN
    DBMS_OUTPUT.PUT_LINE('TRIGGER DE LIBROS EJECUTADO');

    IF :new.precio > 50 THEN
        DBMS_OUTPUT.PUT_LINE('EL PRECIO SE HA REDONDEADO');
        :new.precio := ROUND(:new.precio);
    ELSE
        DBMS_OUTPUT.PUT_LINE('EL PRECIO NO SE HA REDONDEADO');
    END IF;
END;
/

10. Inserccion de empleados

SET SERVEROUTPUT ON;
CREATE OR REPLACE TRIGGER TR_10
BEFORE INSERT ON emple
FOR EACH ROW
DECLARE
    v_conteo NUMBER;
BEGIN
    SELECT COUNT(*) INTO v_conteo 
    FROM emple 
    WHERE emp_no = :new.emp_no;
    IF v_conteo >= 1 THEN
        RAISE_APPLICATION_ERROR(-20001, 'EL NOMBRE DEL EMPLEADO YA EXISTE'); --esto impide meter valores en la tabla
    END IF;
END;
/
