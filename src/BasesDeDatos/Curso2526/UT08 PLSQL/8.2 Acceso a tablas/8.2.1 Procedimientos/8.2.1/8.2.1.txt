CREATE OR REPLACE PROCEDURE insertar_empleados
      (p_emp_no emple.emp_no%TYPE,
       p_apellido emple.apellido%TYPE,
       p_oficio emple.oficio%TYPE,
       p_dir emple.dir%TYPE,
       p_fecha_alta emple.fecha_alt%TYPE,
       p_salario emple.salario%TYPE,
       p_comision emple.comision%TYPE,
       p_dept_no emple.dept_no%TYPE)
   IS
   BEGIN
     INSERT INTO emple VALUES (p_emp_no, p_apellido, p_oficio, p_dir, p_fecha_alta, p_salario, p_comision, p_dept_no);

4. Numero de empleados.

FORMA BASICA
------------

CREATE PROCEDURE NUMEMPLEADOS (p_dept_no emple.dept_no%TYPE)
  IS
    V_num_emp NUMBER(2);
  BEGIN
    SELECT COUNT(*) INTO v_num_emp
    FROM emple
    WHERE dept_no = p_dept_no;
    DBMS_OUTPUT.PUT_LINE('Número de empleados: '|| v_num_emp);
  END;
/

FORMA MEJORADA
--------------

CREATE OR REPLACE
PROCEDURE NUMEMPLEADOS (p_dept_no emple.dept_no%TYPE)
  IS
    V_num_emp NUMBER(2);
  BEGIN
    SELECT COUNT(*) INTO v_num_emp
    FROM emple
    WHERE dept_no = p_dept_no;
    IF v_num_emp = 0
		THEN
			DBMS_OUTPUT.PUT_LINE('NO EXISTE ESE DEPARTAMENTO EN LA TABLA DE EMPLEADOS: ');
		ELSE
			DBMS_OUTPUT.PUT_LINE('Número de empleados: '|| v_num_emp);
    END IF;
  END;
/

5. Estadisticas empleados.

CREATE OR REPLACE PROCEDURE ESTADISTICAS (p_dept_no emple.dept_no%TYPE)
  IS
    V_num_emp NUMBER(2);
  BEGIN
    SELECT COUNT(*) INTO v_num_emp
    FROM emple
    WHERE dept_no = p_dept_no;

	IF v_num_emp = 0 THEN
		DBMS_OUTPUT.PUT_LINE('NO EXISTE ESE DEPARTAMENTO EN LA TABLA DE EMPLEADOS: ');
    ELSIF v_num_emp > 5 THEN
		DBMS_OUTPUT.PUT_LINE(' Hay muchos empleados: '|| v_num_emp);

    ELSIF v_num_emp < 5 THEN
		DBMS_OUTPUT.PUT_LINE('Hay pocos empleados :'|| v_num_emp);
    END IF;
  END;
/

6. Modificar localidad.

CREATE PROCEDURE MODIFICARLOCALIDAD
  (p_dept_no depart.dept_no %TYPE,
   p_loc depart.loc %TYPE)
IS
  BEGIN
    SELECT COUNT(*) INTO v_num_emp
    FROM emple
    WHERE dept_no = p_dept_no;

	IF v_num_emp = 0 THEN
		DBMS_OUTPUT.PUT_LINE('NO EXISTE ESE DEPARTAMENTO  ');
    ELSE
		UPDATE depart
			SET loc = p_loc
			WHERE dept_no = p_dept_no;
	END IF;
  END;
/

CREATE OR REPLACE PROCEDURE EMPLEADOS (APE VARCHAR)
IS
      V_cobran_mas NUMBER(2);
    BEGIN
      SELECT COUNT (*) INTO v_cobran_mas
      FROM emple
      WHERE salario > (SELECT salario
                       FROM emple
                       WHERE apellido=APE);

     DBMS_OUTPUT.PUT_LINE ('Nº de empleados que cobran más que : ' || APE || v_cobran_mas);

8. Oficio mayor.

FORMA 1
-------
CREATE OR REPLACE PROCEDURE OFICIOMAYOR
IS
      v_oficio emple.oficio %TYPE;
    BEGIN
      SELECT oficio INTO v_oficio
      FROM emple
      GROUP BY oficio
      HAVING COUNT(*) = (SELECT MAX(COUNT(*))
                         FROM emple
                         GROUP BY oficio);

	  IF v_oficio='VENDEDOR' THEN
			UPDATE emple
			SET salario=salario+1000
	  ELSIF v_oficio='EMPLEADO' or v_oficio='ANALISTA' THEN
			UPDATE emple
			SET salario=salario+10000
      ELSIF v_oficio='PRESIDENTE' THEN
			UPDATE emple
			SET salario=salario+20000
	  END IF;

	DBMS_OUTPUT.PUT_LINE('  '||v_oficio);
    END;


FORMA 2
-------

CREATE OR REPLACE PROCEDURE OFICIOMAYOR
IS
	NUM NUMBER;
	OFICIO2 VARCHAR2(10);
BEGIN
	SELECT COUNT(OFICIO) INTO NUM
	FROM EMPLE
	WHERE OFICIO =
	(
		SELECT MAX(OFICIO)
		FROM EMPLE
	);

	SELECT OFICIO INTO OFICIO2
	FROM EMPLE
	GROUP BY OFICIO
	HAVING COUNT(OFICIO)=NUM;

	DBMS_OUTPUT.PUT_LINE('TIENE '||NUM||' '||OFICIO2);


END;
/

9. Modificar stock.

DECLARE
   v_min productos.cod_producto%TYPE;
   v_max productos.cod_producto%TYPE;
   BEGIN
     SELECT MIN(cod_producto), MAX(cod_producto) INTO v_min, v_max
     FROM productos;

     FOR v_cont IN v_min .. v_max LOOP
       UPDATE productos
       SET stock = 20
       WHERE cod_producto=v_cont;
     END LOOP;
   END;
