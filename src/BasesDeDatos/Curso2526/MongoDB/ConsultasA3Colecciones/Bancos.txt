6. Mostrar las cuentas de clientes que se apelliden SANCHEZ

-- forma 1

db.clientes.aggregate([
  {
    $match: { apellidos: "SANCHEZ" }
  },
  {
    $lookup: {
      from: "cuentas",
      localField: "codcuenta",
      foreignField: "codcuenta",
      as: "cuenta_info"
    }
  },
  {
    $unwind: "$cuenta_info"
  },
  {
    $project: 	{
			_id:0, codcuenta:"$cuenta_info.codcuenta",nomcuenta:"$cuenta_info.nomcuenta"
		}
  }
])

12. Mostrar las sucursales con al menos un cliente

db.sucursales.aggregate([
  {
    $lookup: {
      from: "cuentas",
      localField: "codsucursal",
      foreignField: "codsucursal",
      as: "cuentas"
    }
  },
  {
    $lookup: {
      from: "clientes",
      localField: "cuentas.codcuenta",
      foreignField: "codcuenta",
      as: "clientes"
    }
  },
  {
    $match: { "clientes.0": { $exists: true } }
  },
  {
    $project: {
      _id: 0,
      nombre: "$nomsucursal"
    }
  }
]);
