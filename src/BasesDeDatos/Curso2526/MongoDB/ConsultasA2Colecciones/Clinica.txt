1. Obtener todos los hospitales y sus médicos

db.hospitales.aggregate([
  {
    $lookup: {
      from: "medicos",
      localField: "_id",
      foreignField: "hospital_id",
      as: "medicos_asignados"
    }
  }
])

2. Contar el número de médicos por hospital

db.medicos.aggregate([
  {
    $group: {
      _id: "$hospital_id",
      totalMedicos: { $sum: 1 }
    }
  }
])

3. Mostrar los hospitales con médicos en Cardiología

db.medicos.aggregate([
  {
    $match: { especialidad: "Cardiología" }
  },
  {
    $lookup: {
      from: "hospitales",
      localField: "hospital_id",
      foreignField: "_id",
      as: "info_hospital"
    }
  },
  {
    $unwind: "$info_hospital"
  },
  {
    $project: {
      _id: 0,
      nombre_hospital: "$info_hospital.nombre",
      medico: "$nombre",
      especialidad: 1
    }
  }
])

4. Obtener nombres y direcciones de hospitales sin médicos

db.hospitales.aggregate([
  {
    $lookup: {
      from: "medicos",
      localField: "_id",
      foreignField: "hospital_id",
      as: "lista_medicos"
    }
  },
  {
    $match: { lista_medicos: { $size: 0 } }
  },
  {
    $project: {
      _id: 0,
      nombre: 1,
      direccion: 1
    }
  }
])

5. Mostrar los médicos por especialidad

db.medicos.aggregate([
  {
    $group: {
      _id: "$especialidad",
      medicos: { $push: "$nombre" },
      total: { $sum: 1 }
    }
  }
])

6. Obtener la información de médicos y su hospital

db.medicos.aggregate([
  {
    $lookup: {
      from: "hospitales",
      localField: "hospital_id",
      foreignField: "_id",
      as: "info_hospital"
    }
  },
  {
    $unwind: "$info_hospital"
  },
  {
    $project: {
      _id: 0,
      medico: "$nombre",
      especialidad: 1,
      hospital: "$info_hospital.nombre",

    }
  }
])

7. Mostrar los médicos que trabajan en hospitales de Madrid

db.medicos.aggregate([
  {
    $lookup: {
      from: "hospitales",
      localField: "hospital_id",
      foreignField: "_id",
      as: "info_hospital"
    }
  },
  {
    $unwind: "$info_hospital"
  },
  {
    $match: { "info_hospital.ciudad": "Madrid" }
  },
  {
    $project: {
      _id: 0,
      nombre: 1,
      especialidad: 1,
      hospital: "$info_hospital.nombre"
    }
  }
])

12. Contar cuántos médicos hay en total por ciudad


db.medicos.aggregate([
  {
    $lookup: {
      from: "hospitales",
      localField: "hospital_id",
      foreignField: "_id",
      as: "info_hospital"
    }
  },
  { $unwind: "$info_hospital" },
  {
    $group: {
      _id: "$info_hospital.ciudad",
      totalMedicos: { $sum: 1 }
    }
  }
])



13. Mostrar las especialidades únicas por ciudad (usar $addset
que lo que hace es quitar duplicados en un array)


db.medicos.aggregate([
  {
    $lookup: {
      from: "hospitales",
      localField: "hospital_id",
      foreignField: "_id",
      as: "info_hospital"
    }
  },
  { $unwind: "$info_hospital" },
  {
    $group: {
      _id: "$info_hospital.ciudad",
      especialidadesUnicas: { $addToSet: "$especialidad" }
    }
  }
])


14. Mostrar todos los hospitales con el número total de
especialidades distintas


db.hospitales.aggregate([
  {
    $lookup: {
      from: "medicos",
      localField: "_id",
      foreignField: "hospital_id",
      as: "cuadro_medico"
    }
  },
  {
    $project: {
      _id: 0,
      hospital: "$nombre",
      totalEspecialidades: { 
        $size: { $setUnion: ["$cuadro_medico.especialidad", []] }
      }
    }
  }
])
