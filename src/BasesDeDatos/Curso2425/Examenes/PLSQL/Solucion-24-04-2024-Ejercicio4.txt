create or replace trigger tr_modificar_tienda_saldo
 
 before insert or update of saldo on centros
 
 for each row
 
 DECLARE NUM number;
 
 begin 
 
  if (:new.saldo<0) then
        RAISE_APPLICATION_ERROR(-20001,'Has introducido un saldo NEGATIVO');
  end if;
  if (:new.saldo=0) then
        RAISE_APPLICATION_ERROR(-20001,'Has introducido un saldo CERO');
  end if;
  
  if (:old.saldo<=30000) and (:new.saldo>30000) then
   delete from beneficios where codigo=:old.codigo;
  end if;
  
  if (:old.saldo>30000) and (:new.saldo<=30000) then
   insert into beneficios values(:new.codigo,:new.saldo);
  end if;
  
  select COUNT(*) INTO NUM 
  from centros
  where codigo=:NEW.codigo;
    if (NUM != 0) then 
      		RAISE_APPLICATION_ERROR(-20001,  'EL CENTRO YA EXISTE');
  end if;

 end;
